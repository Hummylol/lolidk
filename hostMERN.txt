inside backend package.json "type": "module" below  main:index etc etc

inside index.js:

    import express from "express";
    import cors from 'cors'


    const app = express();
    app.use(cors())

    app.get("/",(req,res)=>{
        res.json({message:"Hello"});
    })

    app.listen(8000,()=>{
        console.log("Server started at port 8000");
    })
